python
# BFAST quick test
import numpy as np
from BFAST_Python.bfast_main import BFAST

# Generate test data
np.random.seed(123)
n = 100
time = np.arange(n)
data = np.sin(2 * np.pi * time / 10) + np.random.normal(0, 0.5, n)
data[50:] += 3  # Add breakpoint at position 50

# Run BFAST
model = BFAST(start=0, freq=1, h=0.15)
result = model.fit(data, time)

# Output results
print(f"Detected breakpoints: {result.breaks}")
